Run

```sh
docker volume create uni-diploma_keycloak_data
docker volume create uni-diploma_minio_data
docker volume create uni-diploma_postgres_data

docker run --rm -v uni-diploma_keycloak_data:/data -v "$(pwd):/backup" alpine sh -c "tar xvf /backup/keycloak_backup.tar -C /data"
docker run --rm -v uni-diploma_minio_data:/data -v "$(pwd):/backup" alpine sh -c "tar xvf /backup/minio_backup.tar -C /data"
docker run --rm -v uni-diploma_postgres_data:/data -v "$(pwd):/backup" alpine sh -c "tar xvf /backup/postgres_backup.tar -C /data"
```
